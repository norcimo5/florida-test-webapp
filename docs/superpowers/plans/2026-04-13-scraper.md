```python
import time
import re
import requests
from bs4 import BeautifulSoup
from scraper.models import Question, QuestionEN, QuestionES, Keyword

HEADERS = {
    'User-Agent': 'Mozilla/5.0 (compatible; FL-Driver-Prep-Scraper/1.0; educational use)'
}

CATEGORY_KEYWORDS = {
    'traffic-signals': ['light', 'signal', 'flashing', 'red', 'yellow', 'green'],
    'speed-limits': ['speed', 'mph', 'limit'],
    'school-zones': ['school', 'bus', 'children'],
    'dui': ['alcohol', 'blood', 'bac', 'drunk', 'dui'],
    'road-markings': ['line', 'marking', 'lane', 'solid', 'dashed'],
    'right-of-way': ['yield', 'right of way', 'intersection', 'priority'],
}

def classify_category(question_text: str) -> str:
    text = question_text.lower()
    for category, keywords in CATEGORY_KEYWORDS.items():
        if any(kw in text for kw in keywords):
            return category
    return 'general'


def scrape_flhsmv(urls: list[str] | None = None) -> list[Question]:
    """
    Scrape practice questions from FLHSMV-linked resources.
    Returns a list of Question objects with EN filled, ES=None, keywords=[].
    """
    if urls is None:
        urls = [
            'https://www.eforms.com/dmv/florida/',
            'https://driving-tests.org/florida/',
        ]

    questions: list[Question] = []
    q_index = 1

    for url in urls:
        try:
            resp = requests.get(url, headers=HEADERS, timeout=10)
            resp.raise_for_status()
        except requests.RequestException as e:
            print(f'[flhsmv] Failed to fetch {url}: {e}')
            continue

        soup = BeautifulSoup(resp.text, 'lxml')

        # Most DMV practice sites wrap questions in elements with class
        # containing 'question'. Adjust selector as needed after inspection.
        question_blocks = soup.select('.question, [class*="question-block"], [class*="quiz-question"]')

        for block in question_blocks:
            q_text_el = block.select_one('.question-text, h3, p')
            answer_els = block.select('.answer, li, [class*="choice"]')

            if not q_text_el or len(answer_els) < 2:
                continue

            q_text = q_text_el.get_text(strip=True)
            choices = [el.get_text(strip=True) for el in answer_els if el.get_text(strip=True)]

            # Correct answer: look for aria-correct, data-correct, or 'correct' class
            correct_index = 0
            for i, el in enumerate(answer_els):
                if el.get('data-correct') == 'true' or 'correct' in el.get('class', []):
                    correct_index = i
                    break

            if len(choices) < 2:
                continue

            questions.append(Question(
                id=f'q_{q_index:04d}',
                en=QuestionEN(
                    question=q_text,
                    choices=choices[:4],  # cap at 4 choices
                    correct=correct_index,
                ),
                es=None,
                keywords=[],
                category=classify_category(q_text),
                source='flhsmv-scrape',
            ))
            q_index += 1

        time.sleep(2)  # polite delay between requests

    return questions
```

```python
from rapidfuzz import fuzz
from scraper.models import Question


def normalize(text: str) -> str:
    """Lowercase and strip punctuation for comparison."""
    return ''.join(c.lower() for c in text if c.isalnum() or c.isspace()).strip()


def deduplicate(questions: list[Question], threshold: int = 90) -> list[Question]:
    """
    Remove near-duplicate questions using token sort ratio similarity.
    Keeps the first occurrence. Reassigns IDs sequentially after dedup.
    threshold: minimum similarity score (0-100) to consider a duplicate.
    """
    kept: list[Question] = []
    kept_normalized: list[str] = []

    for q in questions:
        norm = normalize(q.en.question)
        is_dup = any(
            fuzz.token_sort_ratio(norm, existing) >= threshold
            for existing in kept_normalized
        )
        if not is_dup:
            kept.append(q)
            kept_normalized.append(norm)

    # Reassign sequential IDs
    for i, q in enumerate(kept, start=1):
        q.id = f'q_{i:04d}'

    return kept
```

```python
import time
from deep_translator import GoogleTranslator
from scraper.models import Question, QuestionES


def translate_questions(questions: list[Question]) -> list[Question]:
    """
    Add Spanish translations to questions that don't already have one.
    Skips questions where es is already populated.
    Uses Google Translate via deep-translator (free, no API key).
    Adds a 0.5s delay between requests to avoid rate limiting.
    """
    translator = GoogleTranslator(source='en', target='es')

    for q in questions:
        if q.es is not None:
            continue  # already translated — skip

        try:
            es_question = translator.translate(q.en.question)
            time.sleep(0.3)

            es_choices = []
            for choice in q.en.choices:
                es_choice = translator.translate(choice)
                time.sleep(0.2)
                es_choices.append(es_choice)

            q.es = QuestionES(question=es_question, choices=es_choices)

        except Exception as e:
            print(f'[translate] Failed to translate {q.id}: {e}')
            q.es = None  # leave as None; frontend shows fallback

    return questions
```

```python
from deep_translator import GoogleTranslator
from scraper.models import Question, Keyword

# Known important multi-word phrases to extract as keywords
PHRASE_PATTERNS = [
    'flashing yellow', 'flashing red', 'red light', 'yellow light', 'green light',
    'speed limit', 'school zone', 'school bus', 'right of way', 'blood alcohol',
    'BAC', 'solid yellow', 'dashed yellow', 'solid white', 'center line',
    'divided highway', 'pedestrian crosswalk', 'railroad crossing', 'stop sign',
    'yield sign', 'no passing zone', 'work zone', 'construction zone',
    'right turn', 'left turn', 'U-turn', 'three-point turn',
    'following distance', 'blind spot', 'right of way',
]

IMPORTANT_SINGLE_WORDS = [
    'yield', 'merge', 'pedestrian', 'intersection', 'median',
    'overpass', 'ramp', 'acceleration', 'deceleration',
]


def extract_keywords(questions: list[Question]) -> list[Question]:
    """
    Tag each question with key English terms that appear in the question text.
    Translates each keyword to Spanish using Google Translate.
    Questions with no ES translation are skipped (can't build keyword pairs).
    """
    translator = GoogleTranslator(source='en', target='es')

    for q in questions:
        if q.es is None:
            continue

        found_en: list[str] = []
        text_lower = q.en.question.lower()

        # Check multi-word phrases first (higher priority)
        for phrase in PHRASE_PATTERNS:
            if phrase.lower() in text_lower and phrase not in found_en:
                found_en.append(phrase)

        # Check single important words if no phrases found
        if not found_en:
            for word in IMPORTANT_SINGLE_WORDS:
                if word in text_lower:
                    found_en.append(word)

        # Fallback: extract the first noun phrase (3-5 words) from question
        if not found_en:
            words = q.en.question.split()
            # skip question words and take first meaningful chunk
            skip = {'what', 'when', 'where', 'which', 'who', 'how', 'why',
                    'does', 'do', 'is', 'are', 'must', 'should', 'can', 'a', 'an', 'the'}
            meaningful = [w.rstrip('?,') for w in words if w.lower().rstrip('?,') not in skip]
            if meaningful:
                found_en.append(' '.join(meaningful[:3]))

        # Translate each keyword to Spanish
        keywords: list[Keyword] = []
        for en_term in found_en[:3]:  # cap at 3 keywords per question
            try:
                es_term = translator.translate(en_term)
                keywords.append(Keyword(en=en_term, es=es_term))
            except Exception:
                keywords.append(Keyword(en=en_term, es=en_term))  # fallback: use EN

        q.keywords = keywords

    return questions
```

```python
import json
import dataclasses
from pathlib import Path
from scraper.models import Question


def to_dict(q: Question) -> dict:
    return {
        'id': q.id,
        'en': {
            'question': q.en.question,
            'choices': q.en.choices,
            'correct': q.en.correct,
        },
        'es': {
            'question': q.es.question,
            'choices': q.es.choices,
        } if q.es else None,
        'keywords': [{'en': kw.en, 'es': kw.es} for kw in q.keywords],
        'category': q.category,
        'source': q.source,
    }


def validate(questions: list[Question]) -> None:
    """Abort if questions.json would be malformed."""
    for q in questions:
        assert q.id, f'Missing id: {q}'
        assert q.en.question, f'Missing EN question: {q.id}'
        assert len(q.en.choices) >= 2, f'Too few choices: {q.id}'
        assert 0 <= q.en.correct < len(q.en.choices), f'Invalid correct index: {q.id}'
        assert len(q.keywords) >= 1, f'No keywords: {q.id}'


def export_questions(questions: list[Question], output_path: str) -> None:
    """Validate and write questions to output_path as JSON."""
    validate(questions)
    path = Path(output_path)
    path.parent.mkdir(parents=True, exist_ok=True)
    with open(path, 'w', encoding='utf-8') as f:
        json.dump([to_dict(q) for q in questions], f, ensure_ascii=False, indent=2)
    print(f'[export] Wrote {len(questions)} questions to {output_path}')
```

```python
#!/usr/bin/env python3
"""
Full scraper pipeline orchestrator.
Run: python -m scraper.run
"""
import sys
from scraper.sources.flhsmv import scrape_flhsmv
from scraper.sources.dmvorg import scrape_dmvorg
from scraper.pipeline.dedup import deduplicate
from scraper.pipeline.translate import translate_questions
from scraper.pipeline.keywords import extract_keywords
from scraper.pipeline.export import export_questions

OUTPUT_PATH = 'src/data/questions.json'


def run():
    print('[run] Step 1: Scraping FLHSMV...')
    flhsmv_qs = scrape_flhsmv()
    print(f'[run]   Got {len(flhsmv_qs)} questions from FLHSMV')

    print('[run] Step 2: Scraping DMV.org...')
    dmvorg_qs = scrape_dmvorg()
    print(f'[run]   Got {len(dmvorg_qs)} questions from DMV.org')

    print('[run] Step 3: Deduplicating...')
    merged = deduplicate(flhsmv_qs + dmvorg_qs)
    print(f'[run]   {len(merged)} unique questions after dedup')

    print('[run] Step 4: Translating to Spanish...')
    translated = translate_questions(merged)
    translated_count = sum(1 for q in translated if q.es is not None)
    print(f'[run]   {translated_count}/{len(translated)} questions translated')

    print('[run] Step 5: Extracting keywords...')
    with_keywords = extract_keywords(translated)
    keyword_count = sum(1 for q in with_keywords if q.keywords)
    print(f'[run]   {keyword_count}/{len(with_keywords)} questions have keywords')

    print(f'[run] Step 6: Exporting to {OUTPUT_PATH}...')
    try:
        export_questions(with_keywords, OUTPUT_PATH)
    except AssertionError as e:
        print(f'[run] VALIDATION FAILED: {e}')
        sys.exit(1)

    print('[run] Done.')


if __name__ == '__main__':
    run()
```

```yaml
          git add src/data/questions.json
          git diff --staged --quiet || git commit -m "chore: update question bank via scraper [skip ci]"
          git push

      - name: Set up Node for Vercel deploy
        if: ${{ inputs.deploy == 'true' }}
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Deploy to Vercel
        if: ${{ inputs.deploy == 'true' }}
        run: |
          npm install
          npx vercel --prod --token=${{ secrets.VERCEL_TOKEN }} --yes
        env:
          VERCEL_ORG_ID: ${{ secrets.VERCEL_ORG_ID }}
          VERCEL_PROJECT_ID: ${{ secrets.VERCEL_PROJECT_ID }}
```

> - `VERCEL_TOKEN` — from vercel.com/account/tokens
> - `VERCEL_ORG_ID` — from `.vercel/project.json` after first deploy
> - `VERCEL_PROJECT_ID` — from `.vercel/project.json` after first deploy
